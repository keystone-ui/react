---
title: Accordion
description: Vertically stacked expandable and collapsible content sections for organizing dense information.
---

## Import

```tsx
import { Accordion, AccordionItem, AccordionTrigger, AccordionContent } from "keystoneui/accordion";
```

## Usage

```tsx
<Accordion>
  <AccordionItem value="item-1">
    <AccordionTrigger>Section 1</AccordionTrigger>
    <AccordionContent>Content for section 1.</AccordionContent>
  </AccordionItem>
  <AccordionItem value="item-2">
    <AccordionTrigger>Section 2</AccordionTrigger>
    <AccordionContent>Content for section 2.</AccordionContent>
  </AccordionItem>
</Accordion>
```
