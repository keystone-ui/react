---
title: Table
description: Structured data table with rows, columns, headers, and cells for displaying tabular data.
---

## Import

```tsx
import { Table, TableHeader, TableBody, TableRow, TableHead, TableCell } from "keystoneui/table";
```

## Usage

```tsx
<Table>
  <TableHeader>
    <TableRow>
      <TableHead>Name</TableHead>
      <TableHead>Email</TableHead>
    </TableRow>
  </TableHeader>
  <TableBody>
    <TableRow>
      <TableCell>John Doe</TableCell>
      <TableCell>john@example.com</TableCell>
    </TableRow>
  </TableBody>
</Table>
```
