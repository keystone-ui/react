---
title: InputOTP
description: One-time password input with individual digit slots for verification codes.
---

## Import

```tsx
import { InputOTP, InputOTPGroup, InputOTPSlot, InputOTPSeparator } from "keystoneui/input-otp";
```

## Usage

```tsx
<InputOTP maxLength={6}>
  <InputOTPGroup>
    <InputOTPSlot index={0} />
    <InputOTPSlot index={1} />
    <InputOTPSlot index={2} />
  </InputOTPGroup>
  <InputOTPSeparator />
  <InputOTPGroup>
    <InputOTPSlot index={3} />
    <InputOTPSlot index={4} />
    <InputOTPSlot index={5} />
  </InputOTPGroup>
</InputOTP>
```
